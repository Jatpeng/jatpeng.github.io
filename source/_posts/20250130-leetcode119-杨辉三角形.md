    <!-- 给定一个非负索引 rowIndex，返回「杨辉三角」的第 rowIndex 行。

    在「杨辉三角」中，每个数是它左上方和右上方的数的和。
    
    示例 1:

    输入: rowIndex = 3
    输出: [1,3,3,1]
    示例 2:

    输入: rowIndex = 0
    输出: [1]
    示例 3:

    输入: rowIndex = 1
    输出: [1,1]
    

    提示:

    0 <= rowIndex <= 33 -->

```python
#递推解法（空间复杂度O(k^2)）
class Solution:
    def getRow(self, rowIndex: int) -> List[int]:
        C = [[1] * (i + 1) for i in range(rowIndex + 1)]
        for i in range(0, rowIndex + 1):
            for j in range(1, i):
                C[i][j] = C[i - 1][j - 1] + C[i - 1][j]
        return C[rowIndex]

#滚动数组解法（空间复杂度O(k)）   
class Solution:
    def getRow(self, rowIndex: int) -> List[int]:
        cur,pre = [],[]
        for i in range(rowIndex + 1):
            cur = [1] * (i + 1)
            for j in range(1,i):
                cur[j] = pre[j - 1] + pre[j]
            pre = cur
        return pre
```
